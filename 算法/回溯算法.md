//maze整个路径
//pos起始点/位置
//path是路径
//transverse走过的位置 
function rat_in_maze(
    maze,pos = [0,0], path =[[...pos]],transverse = []
){
    const [x,y] = pos
    if(maze[x][y] === 2){// 基础条件 等于2成功
        return path
    }
    // 记录走过的位置 ，由于用的一维数组则有下列
    transverse[x*maze.length+y] = 1
    const choices = [
        [x+1,y],
    ]
}
